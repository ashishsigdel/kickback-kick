Implementing idempotency keys with the response cached and replayed on retry, not just deduped:

```python
class IdempotencyKey(db.Model):
    key = db.Column(db.String, primary_key=True)
    request_hash = db.Column(db.String, nullable=False)   # hash of the request body
    response_body = db.Column(db.JSON, nullable=True)      # filled in once the charge completes
    status = db.Column(db.String, default="processing")    # processing | completed
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
```

```python
@app.route("/v1/charges", methods=["POST"])
def create_charge():
    data = request.get_json()
    key = request.headers.get("Idempotency-Key")
    if not key:
        return jsonify({"error": "Idempotency-Key header required"}), 400

    req_hash = hashlib.sha256(request.data).hexdigest()

    try:
        db.session.add(IdempotencyKey(key=key, request_hash=req_hash))
        db.session.commit()
    except IntegrityError:
        db.session.rollback()
        existing = IdempotencyKey.query.get(key)
        if existing.request_hash != req_hash:
            return jsonify({"error": "Idempotency-Key reused with a different request body"}), 422
        if existing.status == "processing":
            return jsonify({"error": "request still processing, retry shortly"}), 409
        return jsonify(existing.response_body), 200  # replay original response

    charge = payment_gateway.charge(amount=data["amount"], customer_id=data["customer_id"], source=data["source"])
    db.session.add(Charge(customer_id=data["customer_id"], amount=data["amount"],
                           gateway_charge_id=charge.id, request_id=g.request_id))
    response_body = {"id": charge.id}
    IdempotencyKey.query.filter_by(key=key).update({"status": "completed", "response_body": response_body})
    db.session.commit()
    return jsonify(response_body), 201
```

The `key` column as a primary key with the insert-first pattern is the atomicity guarantee — same shape as the DB-level dedup fix that closes webhook races, applied here to a client-initiated request instead of a provider-initiated one. Two concurrent requests with the same key: one wins the insert, the other gets `IntegrityError` and either replays or gets a 409 if it arrived while the first is still mid-flight.
